  {
    \@@_test:n
      {
        \[
          \begin{gathered}
            A\Arrows{#1}{}Z \\ \Arrows{#1}{H} \\ \Arrows{#1}{g} \\ \Arrows{#1}{2}
          \end{gathered}
        \]
      }
  }

\@@_new_test:nn { single }
  {
    % Simple
    \leftarrow     \leftrightarrow     \rightarrow
    \longleftarrow \longleftrightarrow \longrightarrow
    \twoheadleftarrow \twoheadrightarrow
    % Maps
    \mapsfrom \mapsto \longmapsfrom \longmapsto \twoheadmapsfrom \twoheadmapsto
    % Tail
    \lefttail \righttail \leftdbltail \rightdbltail \leftarrowtail \rightarrowtail
    % Comparison
    { \leftarrow\!\!-\!\!\rightarrow }
    { \leftarrowtail \! \leftrightarrow \! \rightarrowtail }
  }
\@@_new_test:nn { not }
  {
    % Bar
    \barleftarrow \rightarrowbar
    \nvleftarrow \nvleftrightarrow \nvrightarrow
    \nVleftarrow \nVleftrightarrow \nVrightarrow
    \nvtwoheadleftarrow \nvtwoheadrightarrow
    \nVtwoheadleftarrow \nVtwoheadrightarrow
    % Tail
    \nvleftarrowtail        \nvrightarrowtail
    \nVleftarrowtail        \nVrightarrowtail
    \nvtwoheadleftarrowtail \nvtwoheadrightarrowtail
    \nVtwoheadleftarrowtail \nVtwoheadrightarrowtail
    % Not
    \nleftarrow \nleftrightarrow \nrightarrow
    % Comparison
    { \barleftarrow \leftrightarrow \rightarrowbar }
    { \mapsfrom \barleftarrow } { \rightarrowbar \mapsto }
    { \nVleftarrow \twoheadleftarrow \twoheadrightarrow \nVrightarrow }
  }

\@@_new_test:nn { half }
  {
    % Simple
    \leftharpoonup   \leftrightharpoonupup     \rightharpoonup
    \leftharpoondown \leftrightharpoondowndown \rightharpoondown
    \leftrightharpoonupdown \leftrightharpoondownup
    % Bar
    \barleftharpoonup \rightharpoonupbar \barleftharpoondown \rightharpoondownbar
    \leftharpoonupbar \barrightharpoonup \leftharpoondownbar \barrightharpoondown
    % Comparison
    { \leftharpoonup \!\! - \!\! \rightharpoonup }
    { \leftarrow \rightharpoonup } { \leftharpoonup \rightarrow }
  }
\@@_new_test:nn { double }
  {
    \Leftarrow     \Leftrightarrow     \Rightarrow
    { \Leftarrow  =      \Rightarrow  }
    { \Lleftarrow \equiv \Rrightarrow }
    \Longleftarrow \Longleftrightarrow \Longrightarrow
    \Mapsfrom \Mapsto \Longmapsfrom \Longmapsto
    \nLeftarrow  \nLeftrightarrow  \nRightarrow
    \nvLeftarrow \nvLeftrightarrow \nvRightarrow
    \rightleftarrows \leftrightarrows \leftleftarrows \rightrightarrows
    \leftthreearrows \rightthreearrows
    \barleftarrowrightarrowbar
  }
\@@_new_test:nn { double-half }
  {
    \leftharpoonsupdown  \rightharpoonsupdown

    \rightharpoonupdash  \dashrightharpoondown

    { \Leftarrow = \rightharpoonsupdown }
